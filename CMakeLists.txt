cmake_minimum_required(VERSION 3.16)

# --- Usage Instructions ---
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# 1. Configure: cmake -B build -G "MinGW Makefiles"
# 2. Build:     cmake --build build
# 3. Flash:     cmake --build build --target flash
# 4. Clean:     cmake --build build --target clean
# 5. Reset:     Remove-Item -Recurse -Force build (PowerShell)
# --------------------------


# --- Toolchain Configuration ---
set(CMAKE_SYSTEM_NAME Generic)
set(CMAKE_SYSTEM_PROCESSOR arm)

# Specify the cross compiler
set(CMAKE_C_COMPILER arm-none-eabi-gcc)
set(CMAKE_CXX_COMPILER arm-none-eabi-g++)
set(CMAKE_ASM_COMPILER arm-none-eabi-gcc)
set(CMAKE_OBJCOPY arm-none-eabi-objcopy)
set(CMAKE_SIZE arm-none-eabi-size)

# Skip compiler test by building a static library instead of an executable during probe
set(CMAKE_TRY_COMPILE_TARGET_TYPE STATIC_LIBRARY)

project(STM32_Learning C CXX ASM)

# --- MCU & Compiler Flags ---
set(CPU_FLAGS "-mcpu=cortex-m0 -mthumb")

# Enable optimization -O0 for Debug, and common warning flags
set(COMMON_FLAGS "${CPU_FLAGS} -O0 -g3 -Wall -fdata-sections -ffunction-sections")

set(CMAKE_C_FLAGS "${COMMON_FLAGS}" CACHE INTERNAL "C Compiler options")
# Disable exceptions and rtti in C++ to reduce binary size (embedded standard)
set(CMAKE_CXX_FLAGS "${COMMON_FLAGS} -fno-exceptions -fno-rtti" CACHE INTERNAL "C++ Compiler options")
set(CMAKE_ASM_FLAGS "${COMMON_FLAGS}" CACHE INTERNAL "ASM Compiler options")

# --- Linker Flags ---
# Linker script generated by CubeMX, located in Board/v1
set(LINKER_SCRIPT ${CMAKE_CURRENT_SOURCE_DIR}/Board/v1/STM32F051R8Tx_FLASH.ld)

set(CMAKE_EXE_LINKER_FLAGS "${CPU_FLAGS} -T \"${LINKER_SCRIPT}\" -Wl,-Map=${PROJECT_NAME}.map,--cref -Wl,--gc-sections --specs=nano.specs --specs=nosys.specs -lc -lm" CACHE INTERNAL "Linker options")

# --- Macros (Defines) ---
add_compile_definitions(
    USE_HAL_DRIVER
    STM32F051x8
)

# --- Include Directories ---
include_directories(
    Application
    Application/Interfaces
    Board/v1
    Board/v1/CubeMX_Generated/Core/Inc
    ThirdParty/STM32F0xx_HAL_Driver/Inc
    ThirdParty/STM32F0xx_HAL_Driver/Inc/Legacy
    ThirdParty/CMSIS/Device/ST/STM32F0xx/Include
    ThirdParty/CMSIS/Include
    Drivers/Peripherals/inc
    .
)

# --- Source Files ---
# 1. CubeMX Sources (HAL, System, etc.)
file(GLOB_RECURSE CUBEMX_SOURCES
    "Board/v1/CubeMX_Generated/Core/Src/*.c"
    "ThirdParty/STM32F0xx_HAL_Driver/Src/*.c"
)
# Exclude HAL template files (as they cause compilation errors)
list(FILTER CUBEMX_SOURCES EXCLUDE REGEX ".*_template\\.c$")


# IMPORTANT TRICK:
# CubeMX generates main.c which contains main(). But we want to use main() in Application/main.cpp.
# Therefore, we rename main() in CubeMX's main.c to cubemx_main via compile flag.
# This way, we avoid "multiple definition of main" while still leveraging SystemClock_Config() and Error_Handler() inside main.c.
set_source_files_properties(Board/v1/CubeMX_Generated/Core/Src/main.c PROPERTIES COMPILE_DEFINITIONS "main=cubemx_main")

# 2. C++ Sources (Application, Board, Drivers)
file(GLOB_RECURSE CXX_SOURCES
    "Application/*.cpp"
    "Board/v1/*.cpp"
    "Drivers/Peripherals/src/*.cpp"
)

# 3. ASM Sources (Startup file)
set(ASM_SOURCES
    "Board/v1/startup_stm32f051x8.s"
)

# --- Tạo Executable ---
add_executable(${PROJECT_NAME}.elf ${CUBEMX_SOURCES} ${CXX_SOURCES} ${ASM_SOURCES})

# --- Post-processing (Generate .bin, .hex and print size) ---
add_custom_command(TARGET ${PROJECT_NAME}.elf POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O ihex $<TARGET_FILE:${PROJECT_NAME}.elf> ${PROJECT_NAME}.hex
    COMMAND ${CMAKE_OBJCOPY} -O binary $<TARGET_FILE:${PROJECT_NAME}.elf> ${PROJECT_NAME}.bin
    COMMAND ${CMAKE_SIZE} $<TARGET_FILE:${PROJECT_NAME}.elf>
)

# --- Flash Target (OpenOCD) ---
# Default uses ST-Link and STM32F0 series.
# You can change interface/stlink.cfg to jlink.cfg if using J-Link.
add_custom_target(flash
    COMMAND openocd -f interface/stlink.cfg -f target/stm32f0x.cfg -c "program ${PROJECT_NAME}.elf verify reset exit"
    DEPENDS ${PROJECT_NAME}.elf
    COMMENT "Flashing code to chip using OpenOCD..."
)
